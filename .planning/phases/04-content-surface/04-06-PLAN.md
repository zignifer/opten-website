    ```
      const { slug } = useParams<{ slug: string }>();
      const { lang } = useLang();
      const t = useT();
      const data = slug ? guidesBySlug[slug]?.[lang] : undefined;
      if (!data) { return <NotFoundFallback />; }
      return (
        <div className="...dark theme + max-w-800px column...">
          <header>...LegalLayout-style header chrome with brand + LangSwitcher + back-to-home link...</header>
          <main>
            <h1>{data.title}</h1>
            <p className="...intro...">{data.intro}</p>
            <ol className="howto-steps">
              {data.steps.map((step, i) => (
                <li key={i}>
                  <h3>{step.title}</h3>
                  <p>{step.body}</p>
                  {step.before && step.after && (
                    <div className="before-after-grid">
                      <pre>{step.before}</pre>
                      <pre>{step.after}</pre>
                    </div>
                  )}
                  {step.imageSrc && <img src={step.imageSrc} alt={step.title} loading="lazy" width="800" height="450" />}
                </li>
              ))}
            </ol>
            <FaqBlock items={data.faq} headingKey="guide.faqHeading" id="guide-faq" />
          </main>
          <footer>...LegalLayout-style footer chrome...</footer>
        </div>
      );
    }
    ```
